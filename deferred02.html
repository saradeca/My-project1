$.get = function(url, success) {
    var value;
    if (url.indexOf("geturl") != -1) value = "webservice";
    if (url.indexOf("webservice") != -1) value = "getfinaldata";
    if (url.indexOf("getfinaldata") != -1) value = "--risultato finale--";
    return $.Deferred(function (d) { 
        setTimeout(function() { d.resolve(value); }, 100);
    }).done( function() { success(value); }); 
};
